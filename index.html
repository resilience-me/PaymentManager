Simple crowd-fund

/* simple contract to log who has contributed to a project, and to return funds manually */

contract simpleCrowdSale { 

    address owner;
    address public BTCaddress;

    struct Funders {
        string username;
        string returnAddress;
        string txID;
        string returnTX; //if cancelled
    }

    mapping (address => Funders) public funders;


    function simpleCrowdSale(){
        BTCaddress;
    }

    function Fund(address v, string username, string returnAddress, string txID){
        if(msg.sender != owner)
            throw;
        /* funders */
        funders[v].username = username;
        funders[v].returnAddress = returnAddress;
        funders[v].txID = txID;
    }

    function GetCrowdSaleAddress() returns(address){
        return BTCaddress;
    }    

    function getFunder(address v) internal returns(Funders){
        return funders[v];
    }    

    function returnFunds(address v, string returnTX) internal returns(Funders){
        if(msg.sender != owner)
            throw;
        funders[v].returnTX = returnTX;
        return funders[v];
    }    

}
